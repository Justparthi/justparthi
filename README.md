// The evolution of a programmer
function myJourney() {
  const stages = [
    "Hello World 👶",
    "Why doesn't this work?! 😡",
    "Oh, I see the problem 🧐",
    "It works but I don't know why 🤔",
    "Refactor, optimize, document 📝",
    "Mentor others, contribute to open source 🚀"
  ];
  
  let currentStage = stages[Math.floor(Math.random() * stages.length)];
  const coffee = Array(Math.floor(Math.random() * 5) + 1).fill("☕").join("");
  
  return `Currently at: "${currentStage}" with ${coffee} by my side.`;
}

// Output changes every time you refresh!
console.log(myJourney());
